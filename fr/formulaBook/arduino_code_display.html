// ── Buzzer functions ──────────────────────────────────────────────────────────────
// Uses tone()-style PWM via ledcWrite on ESP32-C3
void buzzInit() {
  ledcAttach(BUZZER, 2000, 8);  // pin, freq, resolution — new API (core v3+)
  ledcWriteTone(BUZZER, 0);
}

void buzzTone(int freq, int durationMs) {
  ledcWriteTone(BUZZER, freq);
  delay(durationMs);
  ledcWriteTone(BUZZER, 0);
}

void buzzOff() {
  ledcWriteTone(BUZZER, 0);
}

// Sound effects — non-blocking would need timers, these are short so blocking is fine
void sndMove() { buzzTone(1200, 12); }
void sndRotate() { buzzTone(1000, 10); delay(10); buzzTone(1400, 10); }
void sndLand() { buzzTone(300, 25); }

void sndClearLine(int count) {
  // More lines = more exciting melody
  if (count == 1) {
    buzzTone(800, 40); buzzTone(1000, 40); buzzTone(1200, 60);
  } else if (count == 2) {
    buzzTone(800, 30); buzzTone(1000, 30); buzzTone(1200, 30); buzzTone(1400, 60);
  } else if (count == 3) {
    buzzTone(800, 25); buzzTone(1000, 25); buzzTone(1200, 25); buzzTone(1400, 25); buzzTone(1600, 60);
  } else {  // Tetris! (4 lines)
    buzzTone(800, 20); buzzTone(1000, 20); buzzTone(1200, 20); buzzTone(1400, 20); buzzTone(1600, 20); buzzTone(2000, 80);
  }
}

void sndLevelUp() {
  buzzTone(600, 30); buzzTone(800, 30); buzzTone(1000, 30); buzzTone(1200, 80);
}

void sndGameOver() {
  buzzTone(600, 80); delay(20); buzzTone(500, 80); delay(20); buzzTone(400, 80); delay(20); buzzTone(300, 150);
}

void sndMenuClick() { buzzTone(900, 15); }
void sndSelect() { buzzTone(800, 20); delay(10); buzzTone(1200, 30); }
void sndExit() { buzzTone(600, 20); delay(10); buzzTone(400, 40); }

#define T_COLS 10
#define T_ROWS 18
#define T_SIZE 7  // slightly bigger cells on 160px wide screen
#define T_XOFF 6
#define T_YOFF 4
#define PANEL_X 82  // score panel X (board ends at 6+10*7=76, panel at 82)

uint8_t board[T_ROWS][T_COLS];

uint16_t pieceColors[] = {
  BLACK,
  0x07FF,  // cyan I
  0xFFE0,  // yellow O
  0xF81F,  // purple T
  0x001F,  // blue J
  0xFDA0,  // orange L
  0x07E0,  // green S
  0xF800,  // red Z
};

const uint8_t PIECES[7][4][4] = {
  {{ 0b0000,0b1111,0b0000,0b0000 },{ 0b0010,0b0010,0b0010,0b0010 },{ 0b0000,0b1111,0b0000,0b0000 },{ 0b0010,0b0010,0b0010,0b0010 }},
  {{ 0b0110,0b0110,0b0000,0b0000 },{ 0b0110,0b0110,0b0000,0b0000 },{ 0b0110,0b0110,0b0000,0b0000 },{ 0b0110,0b0110,0b0000,0b0000 }},
  {{ 0b0100,0b1110,0b0000,0b0000 },{ 0b0100,0b0110,0b0100,0b0000 },{ 0b1110,0b0100,0b0000,0b0000 },{ 0b0100,0b1100,0b0100,0b0000 }},
  {{ 0b1000,0b1110,0b0000,0b0000 },{ 0b0110,0b0100,0b0100,0b0000 },{ 0b1110,0b0010,0b0000,0b0000 },{ 0b0100,0b0100,0b1100,0b0000 }},
  {{ 0b0010,0b1110,0b0000,0b0000 },{ 0b0100,0b0100,0b0110,0b0000 },{ 0b1110,0b1000,0b0000,0b0000 },{ 0b1100,0b0100,0b0100,0b0000 }},
  {{ 0b0110,0b1100,0b0000,0b0000 },{ 0b1000,0b1100,0b0100,0b0000 },{ 0b0110,0b1100,0b0000,0b0000 },{ 0b1000,0b1100,0b0100,0b0000 }},
  {{ 0b1100,0b0110,0b0000,0b0000 },{ 0b0100,0b1100,0b1000,0b0000 },{ 0b1100,0b0110,0b0000,0b0000 },{ 0b0100,0b1100,0b1000,0b0000 }},
};

int curPiece, curRot, curX, curY;
uint32_t score = 0;
int level = 1;
int lines = 0;
bool gameOver = false;
int prevLevel = 1;

unsigned long lastMove = 0;

unsigned long lastExitPress = 0;

#define NORMAL_DROP 500
#define SOFT_DROP 80
#define MOVE_REPEAT 120
#define ROT_DEBOUNCE 180
#define EXIT_DEBOUNCE 300

int dropInterval = NORMAL_DROP;

void drawCell(int col, int row, uint8_t ci) {
  int px = T_XOFF + col*T_SIZE;
  int py = T_YOFF + row*T_SIZE;
  if (ci == 0) {
    tft.fillRect(px, py, T_SIZE-1, T_SIZE-1, BLACK);
  } else {
    tft.fillRect(px, py, T_SIZE-1, T_SIZE-1, pieceColors[ci]);
    tft.drawRect(px, py, T_SIZE-1, T_SIZE-1, WHITE);
  }
}

void drawPiece(int px, int py, int piece, int rot, bool erase) {
  for (int r=0; r<4; r++) {
    uint8_t row = PIECES[piece][rot][r];
    for (int c=0; c<4; c++) {
      if (row & (0b1000>>c)) {
        int bx=px+c, by=py+r;
        if (by>=0 && by<T_ROWS && bx>=0 && bx<T_COLS)
          drawCell(bx, by, erase ? 0 : (piece+1));
      }
    }
  }
}

bool collides(int px, int py, int piece, int rot) {
  for (int r=0; r<4; r++) {
    uint8_t row = PIECES[piece][rot][r];
    for (int c=0; c<4; c++) {
      if (row & (0b1000>>c)) {
        int bx=px+c, by=py+r;
        if (bx<0 || bx>=T_COLS || by>=T_ROWS) return true;
        if (by>=0 && board[by][bx]) return true;
      }
    }
  }
  return false;
}

void lockPiece() {
  for (int r=0; r<4; r++) {
    uint8_t row = PIECES[curPiece][curRot][r];
    for (int c=0; c<4; c++) {
      if (row & (0b1000>>c)) {
        int bx=curX+c, by=curY+r;
        if (by>=0 && by<T_ROWS && bx>=0 && bx<T_COLS)
          board[by][bx] = curPiece+1;
      }
    }
  }
}

void redrawBoard() {
  for (int r=0; r<T_ROWS; r++)
    for (int c=0; c<T_COLS; c++)
      drawCell(c, r, board[r][c]);
}

void clearLines() {
  int cleared = 0;
  for (int r=T_ROWS-1; r>=0; r--) {
    bool full = true;
    for (int c=0; c<T_COLS; c++)
      if (!board[r][c]) { full=false; break; }
    if (full) {
      cleared++;
      for (int rr=r; rr>0; rr--)
        for (int c=0; c<T_COLS; c++)
          board[rr][c] = board[rr-1][c];
      for (int c=0; c<T_COLS; c++)
        board[0][c]=0;
      r++;
    }
  }
  if (cleared > 0) {
    int pts[] = {0, 100, 300, 500, 800};
    score += (uint32_t)pts[min(cleared,4)] * level;
    lines += cleared;
    int newLevel = lines/10 + 1;
    if (newLevel > level) {
      level = newLevel;
      dropInterval = max(NORMAL_DROP - (level-1)*40, 100);
      sndLevelUp();  // level up sound
    } else {
      sndClearLine(cleared);  // line clear sound
    }
    redrawBoard();
    updateScore();
  }
}

void spawnPiece() {
  curPiece = random(7);
  curRot = 0;
  curX = T_COLS/2 - 2;
  curY = 0;
  if (collides(curX, curY, curPiece, curRot)) {
    gameOver = true;
    sndGameOver();
    tft.fillRect(T_XOFF, T_YOFF+T_ROWS*T_SIZE/2-18, T_COLS*T_SIZE, 38, BLACK);
    tft.setTextColor(0xF800, BLACK);
    tft.setTextSize(1);
    tft.setCursor(T_XOFF+4, T_YOFF+T_ROWS*T_SIZE/2-10);
    tft.print("GAME OVER");
    tft.setTextColor(WHITE, BLACK);
    tft.setCursor(T_XOFF+1, T_YOFF+T_ROWS*T_SIZE/2+2);
    tft.print("ROT=restart");
    tft.setCursor(T_XOFF+1, T_YOFF+T_ROWS*T_SIZE/2+12);
    tft.print("EXIT=menu");
  }
}

void initTetris() {
  memset(board, 0, sizeof(board));
  score=0; level=1; lines=0; prevLevel=1;
  dropInterval = NORMAL_DROP;
  gameOver = false;
  tft.fillScreen(BLACK);
  tft.drawRect(T_XOFF-1, T_YOFF-1, T_COLS*T_SIZE+2, T_ROWS*T_SIZE+2, DGRAY);
  redrawBoard();
  updateScore();
  spawnPiece();
  if (!gameOver) drawPiece(curX, curY, curPiece, curRot, false);
  lastDrop = millis();
}

void runTetris() {
  if (gameOver) return;
  unsigned long now = millis();
  bool dropHeld = (digitalRead(BTN_DROP) == LOW);
  int interval = dropHeld ? SOFT_DROP : dropInterval;
  
  if (now - lastDrop >= (unsigned long)interval) {
    lastDrop = now;
    drawPiece(curX, curY, curPiece, curRot, true);
    if (!collides(curX, curY+1, curPiece, curRot)) {
      curY++;
      drawPiece(curX, curY, curPiece, curRot, false);
    } else {
      drawPiece(curX, curY, curPiece, curRot, false);
      lockPiece();
      sndLand();
      clearLines();
      spawnPiece();
      if (!gameOver) drawPiece(curX, curY, curPiece, curRot, false);
    }
  }
}

void handleTetrisButtons() {
  if (gameOver) return;
  unsigned long now = millis();
  bool left = (digitalRead(BTN_LEFT) == LOW);
  bool right= (digitalRead(BTN_RIGHT) == LOW);
  bool rot = (digitalRead(BTN_ROT) == LOW);
  
  // Left / Right with repeat
  if ((left||right) && now-lastMove >= MOVE_REPEAT) {
    lastMove = now;
    drawPiece(curX, curY, curPiece, curRot, true);
    if (left && !collides(curX-1, curY, curPiece, curRot)) {
      curX--; sndMove();
    }
    if (right && !collides(curX+1, curY, curPiece, curRot)) {
      curX++; sndMove();
    }
    drawPiece(curX, curY, curPiece, curRot, false);
  }
  
  // Rotate with wall kick
  if (rot && now-lastRotPress >= ROT_DEBOUNCE) {
    lastRotPress = now;
    int nr = (curRot+1)%4;
    bool rotated = false;
    int kicks[] = {0, 1, -1, 2, -2};  // try center, right, left, far right, far left
    for (int k : kicks) {
      if (!collides(curX+k, curY, curPiece, nr)) {
        drawPiece(curX, curY, curPiece, curRot, true);
        curX += k;
        curRot = nr;
        drawPiece(curX, curY, curPiece, curRot, false);
        sndRotate();
        rotated = true;
        break;
      }
    }
  }
}

int drawDelta(int x, int y, uint16_t color) {
  int w=7, h=7;
  tft.drawLine(x+w/2,y-h,x,y,color);
  tft.drawLine(x+w/2,y-h,x+w,y,color);
  tft.drawLine(x,y,x+w,y,color);
  return x+w+1;
}

void drawFormulaPart(int x, int y, const char* text, uint16_t color, uint16_t bg) {
  int i=0;
  while (text[i]!='\0') {
    if (text[i]=='#') {
      x=drawDelta(x,y,color);
      i++;
    } else if (text[i]=='^') {
      i++;
      tft.setTextColor(color,bg);
      while (text[i] && text[i]!=' ' && text[i]!='+' && text[i]!='-' && text[i]!='=' && text[i]!=')' && text[i]!='(') {
        tft.setTextSize(1);
        tft.setCursor(x,y-4);
        tft.print(text[i]);
        x+=5;
        i++;
      }
    } else {
      tft.setTextSize(1);
      tft.setTextColor(color,bg);
      tft.setCursor(x,y);
      tft.print(text[i]);
      x+=6;
      i++;
    }
  }
}

void loop() {
  if (appState == TETRIS) {
    runTetris();
    if (!gameOver) handleTetrisButtons();
    
    // EXIT button — leave tetris any time
    unsigned long now = millis();
    if (digitalRead(BTN_EXIT)==LOW && now-lastExitPress > EXIT_DEBOUNCE) {
      lastExitPress = now;
      sndExit();
      appState = MENU;
      drawMenuFull();
      return;
    }
    
    // Game over: ROT=restart, EXIT=menu
    if (gameOver) {
      if (digitalRead(BTN_ROT)==LOW && now-lastRotPress > 300) {
        lastRotPress = now;
        initTetris();
      }
    }
    delay(10);
    return;
  }
  
  handleMenuButtons();
  delay(10);
}

unsigned long menuLastPress = 0;
#define MENU_DEBOUNCE 150

void handleMenuButtons() {
  unsigned long now = millis();
  if (now - menuLastPress < MENU_DEBOUNCE) return;
  
  bool left = (digitalRead(BTN_LEFT) == LOW);
  bool right = (digitalRead(BTN_RIGHT) == LOW);
  bool rot = (digitalRead(BTN_ROT) == LOW);
  bool drop = (digitalRead(BTN_DROP) == LOW);
  bool ex = (digitalRead(BTN_EXIT) == LOW);
  
  if (!left && !right && !rot && !drop && !ex) return;
  
  menuLastPress = now;
  
  if (appState == MENU) {
    if (left) {
      int p=menuIndex;
      menuIndex=(menuIndex-1+MENU_COUNT)%MENU_COUNT;
      updateMenuRow(p);
      updateMenuRow(menuIndex);
      sndMenuClick();
    }
    if (right) {
      int p=menuIndex;
      menuIndex=(menuIndex+1)%MENU_COUNT;
      updateMenuRow(p);
      updateMenuRow(menuIndex);
      sndMenuClick();
    }
    if (rot) {
      sndSelect();
      if (menuIndex==3) {
        appState=TETRIS;
        initTetris();
      } else {
        enterSubject();
        drawFormulaFull();
      }
    }
  } else {  // SUBJECT
    if (left) {
      formulaIndex=(formulaIndex-1+formulaCount)%formulaCount;
      updateFormulaContent();
      sndMenuClick();
    }
    if (right) {
      formulaIndex=(formulaIndex+1)%formulaCount;
      updateFormulaContent();
      sndMenuClick();
    }
    if (drop || ex) {
      sndExit();
      appState=MENU;
      drawMenuFull();
    }
  }
}

// ── Menu ──────────────────────────────────────────────────────────────────────────

void drawMenuFull() {
  tft.fillScreen(BLACK);
  tft.fillRect(0,0,160,18,WHITE);
  tft.setTextColor(BLACK);
  tft.setTextSize(1);
  tft.setCursor(24,6);
  tft.print("SELECT SUBJECT");
  for (int i=0; i<MENU_COUNT; i++) updateMenuRow(i);
}

void updateMenuRow(int i) {
  int y = 22+i*26;
  Theme& t = themes[menuTheme[i]];
  tft.fillRect(4,y,152,22,BLACK);
  if (i==menuIndex) {
    tft.fillRoundRect(4,y,152,22,4,t.bar);
    tft.setTextColor(WHITE,t.bar);
  } else {
    tft.drawRoundRect(4,y,152,22,4,t.bar);
    tft.setTextColor(t.bar,BLACK);
  }
  tft.setTextSize(1);
  tft.setCursor(12,y+7);
  tft.print(menuItems[i]);
}

void enterSubject() {
  formulaIndex=0;
  appState=SUBJECT;
  switch (menuIndex) {
    case 0: currentFormulas=mathFormulas; formulaCount=MATH_COUNT; break;
    case 1: currentFormulas=physicsFormulas; formulaCount=PHYSICS_COUNT; break;
    case 2: currentFormulas=beFormulas; formulaCount=BE_COUNT; break;
  }
}

// ── Formula ───────────────────────────────────────────────────────────────────────

void drawFormulaFull() {
  Formula& f=currentFormulas[formulaIndex];
  Theme& t=themes[f.style];
  tft.fillScreen(t.bg);
  tft.fillRect(0,0,160,20,t.bar);
  tft.fillRect(0,20,160,2,t.accent);
  tft.setTextSize(1);
  tft.setTextColor(t.barText,t.bar);
  tft.setCursor(4,7);
  tft.print(f.title);
  drawFormulaContent(f,t);
  prevStyle=f.style;
}

void updateFormulaContent() {
  Formula& f=currentFormulas[formulaIndex];
  Theme& t=themes[f.style];
  if (f.style!=prevStyle) {
    drawFormulaFull();
    return;
  }
  tft.fillRect(0,0,160,20,t.bar);
  tft.setTextSize(1);
  tft.setTextColor(t.barText,t.bar);
  tft.setCursor(4,7);
  tft.print(f.title);
  tft.fillRect(0,22,160,96,t.bg);
  drawFormulaContent(f,t);
}

void drawFormulaContent(Formula& f, Theme& t) {
  bool has2 = f.line2[0]!='\0';
  bool has3 = f.line3[0]!='\0';
  if (!has2 && !has3) {
    drawFormulaPart(4, 62, f.line1, t.formula, t.bg);
  } else if (has2 && !has3) {
    drawFormulaPart(4, 48, f.line1, t.formula, t.bg);
    drawFormulaPart(4, 68, f.line2, t.formula, t.bg);
  } else {
    drawFormulaPart(4, 38, f.line1, t.formula, t.bg);
    drawFormulaPart(4, 56, f.line2, t.formula, t.bg);
    drawFormulaPart(4, 74, f.line3, t.formula, t.bg);
  }
  
  char pg[8];
  snprintf(pg,sizeof(pg),"%d/%d",formulaIndex+1,formulaCount);
  tft.setTextSize(1);
  tft.setTextColor(t.accent,t.bg);
  tft.setCursor(160-strlen(pg)*6-3, 120);
  tft.print(pg);
}
